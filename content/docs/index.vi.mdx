## Bắt đầu nhanh

<Tabs items={["pnpm", "npm", "yarn"]}>
  <Tab value="pnpm">
    ```bash
    pnpm create next-app --example doculite-docs-template my-docs
    cd my-docs
    pnpm dev
    ```
  </Tab>
  <Tab value="npm">
    ```bash
    npx create-next-app --example doculite-docs-template my-docs
    cd my-docs
    npm run dev
    ```
  </Tab>
  <Tab value="yarn">
    ```bash
    yarn create next-app --example doculite-docs-template my-docs
    cd my-docs
    yarn dev
    ```
  </Tab>
</Tabs>
